37 void PadCalibCCDBBuilder::CheckIfIsolatedHotPadCandidate(TH2F* hDet, std::vector<int> coordinates)
38 {
39  if ((int)coordinates.size() != 4) {
40  std::cerr << "Invalid size of the coordinates vector!" << std::endl;
41  return;
42  }
43  float averageGain = ComputeDetectorAverage(hDet);
44 
45  if (hDet->GetBinContent(coordinates[0], coordinates[1]) > 2 * averageGain) {
46  int sizeOfHotArea = IsolatedHotPadsContainmentSize(hDet, coordinates[0], coordinates[1]);
47  if (sizeOfHotArea > 0 && sizeOfHotArea < 4) {
48  ReplaceIsolatedHotPads(hDet, coordinates[0], coordinates[1], sizeOfHotArea);
49  }
50  } else if (hDet->GetBinContent(coordinates[0], coordinates[1]) == 0 && hDet->GetBinContent(coordinates[2], coordinates[3]) != 0) {
51  int sizeOfHotArea = IsolatedHotPadsContainmentSize(hDet, coordinates[2], coordinates[3]);
52  if (sizeOfHotArea == 1) {
53  ReplaceIsolatedHotPads(hDet, coordinates[2], coordinates[3], sizeOfHotArea);
54  }
55  }

58 void PadCalibCCDBBuilder::CheckIfSmallerCloserToCenter(TH2F* hDet, std::vector<int> coordinates, float allowedDifference)
59 {
60  if (hDet->GetBinContent(coordinates[0], coordinates[1]) == 0 || hDet->GetBinContent(coordinates[2], coordinates[3]) == 0)
61  return;
62 
63  float xCenter = hDet->GetNbinsX() / 2.;
64  float yCenter = hDet->GetNbinsY() / 2.;
65 
66  std::vector<float> vCenter, vPad1, vPad2;
67  vCenter.push_back(hDet->GetNbinsX() / 2.);
68  vCenter.push_back(hDet->GetNbinsY() / 2.);
69 
70  vPad1.push_back(hDet->GetXaxis()->GetBinCenter(coordinates[0]));
71  vPad1.push_back(hDet->GetYaxis()->GetBinCenter(coordinates[1]));
72 
73  vPad2.push_back(hDet->GetXaxis()->GetBinCenter(coordinates[2]));
74  vPad2.push_back(hDet->GetYaxis()->GetBinCenter(coordinates[3]));
75 
76  float dist1 = ComputeDistance(vPad1, vCenter);
77  float dist2 = ComputeDistance(vPad2, vCenter);
78 
79  if ((dist1 < dist2) && ((hDet->GetBinContent(coordinates[2], coordinates[3]) - hDet->GetBinContent(coordinates[0], coordinates[1])) > allowedDifference)) {
80  ReplaceBinCloserToCenter(hDet, coordinates[0], coordinates[1]);
81  }
82 
83  if ((dist1 > dist2) && ((hDet->GetBinContent(coordinates[0], coordinates[1]) - hDet->GetBinContent(coordinates[2], coordinates[3])) > allowedDifference)) {
84  ReplaceBinCloserToCenter(hDet, coordinates[2], coordinates[3]);
85  }
86 }

88 std::vector<int> PadCalibCCDBBuilder::CompareGain(TH2F* hDet, int column, int row, int shiftcolumn, int shiftrow, float allowedDifference)
89 {
90  std::vector<int> coordinates = {-1, -1, -1, -1};
91 
92  int colMax = hDet->GetNbinsX();
93  int rowMax = hDet->GetNbinsY();
94 
95  // only allow shift along one axis and only by one pad
96  if (!(shiftcolumn == 1 && shiftrow == 0) && !(shiftcolumn == 0 && shiftrow == 1))
97  return coordinates;
98 
99  // cheks that the pad is valid
100  if ((column >= colMax) || (row >= rowMax))
101  return coordinates;
102 
103  if ((column == colMax && shiftcolumn == 1) || (row == rowMax && shiftrow == 1))
104  return coordinates; // do not compare with overflow
105 
106  float gain1 = hDet->GetBinContent(column, row);
107  float gain2 = hDet->GetBinContent(column + shiftcolumn, row + shiftrow);
108 
109  if ((gain1 == 0 && gain2 > 0) || (gain1 > 0 && gain2 == 0) || (abs(gain1 - gain2) > allowedDifference)) {
110  coordinates[0] = column;
111  coordinates[1] = row;
112  coordinates[2] = column + shiftcolumn;
113  coordinates[3] = row + shiftrow;
114  }
115 
116  return coordinates;
117 }

119 float PadCalibCCDBBuilder::ComputeDetectorAverage(TH2F* hDet)
120 { // computes an average
121 
122  float average = 0.;
123  int nBinsUsed = 0;
124  for (int irow = 0; irow < hDet->GetNbinsY(); irow++) {
125  for (int icol = 0; icol < hDet->GetNbinsX(); icol++) {
126  float currentGain = hDet->GetBinContent(hDet->GetXaxis()->FindBin(icol), hDet->GetYaxis()->FindBin(irow));
127  if (currentGain == 0)
128  continue;
129  average += currentGain;
130  nBinsUsed++;
131  }
132  }
133  average /= nBinsUsed;
134 
135  return average;
136 }

138 float PadCalibCCDBBuilder::ComputeDistance(std::vector<float> pad1, std::vector<float> pad2)
139 {
140  float distance = -1.;
141 
142  if (pad1.size() != pad2.size()) {
143  std::cerr << "Something fishy with the pad coordinates!" << std::endl;
144  return distance;
145  }
146 
147  for (int i = 0; i < (int)pad1.size(); i++) {
148  distance += TMath::Power(pad1[i] - pad2[i], 2);
149  }
150  distance = TMath::Sqrt(distance);
151 
152  return distance;
153 }

192 TH2F* PadCalibCCDBBuilder::FillTheMap(TH2F* hDet, TString sNewName, int nbuffer)
193 { // clones the map and fills the clone
194  // will fill any map with at least 1 hit!
195  // the map is cloned to limit propagation of neighboring bin content
196  if (sNewName == "") { // create default name
197  sNewName = hDet->GetName();
198  sNewName += "_filled";
199  }
200  TH2F* hDetFilled = (TH2F*)hDet->Clone(sNewName);
201 
202  if (!hDet || hDet->GetEntries() == 0)
203  return hDetFilled;
204 
205  TH2F* hDetTemp = (TH2F*)hDet->Clone("hDetTemp"); // use as intermediate det in the filling process
206  // find empty bins
207  std::vector<std::vector<int>> emptyBinsColRow = FindEmpty(hDetTemp);
208  // loop over empty bins of th clone and fill them with the average gain
209  // calculated from the gain in neighboring bins in the "Temp" map
210  int nEmptyBins = emptyBinsColRow.size();
211 
212  while (nEmptyBins != 0) {
213  for (int ibin = 0; ibin < nEmptyBins; ibin++) {
214  float averageGain = GetAverageFromNeighbors(hDetTemp, emptyBinsColRow[ibin][0], emptyBinsColRow[ibin][1], nbuffer);
215  FillInTheGap(hDetFilled, emptyBinsColRow[ibin][0], emptyBinsColRow[ibin][1], averageGain);
216  }
217  emptyBinsColRow.clear();
218  hDetTemp = (TH2F*)hDetFilled->Clone("hDetTemp");
219  emptyBinsColRow = FindEmpty(hDetTemp);
220  nEmptyBins = emptyBinsColRow.size();
221  } // will continue the loop till all bins are filled
222 
223  return hDetFilled;
224 }

287 float PadCalibCCDBBuilder::GetAverageFromNeighbors(TH2F* hDet, int column, int row, int nbuffer)
288 { // takes bin (column,row) from hDet map
289  // and averages the gain in its (up to) 8 neighbours
290  float average = 0.;
291  std::vector<float> gainNeighbor;
292  int offset = nbuffer / 2;
293 
294  for (int irow = 0; irow < nbuffer; irow++) {
295 
296  int rowNeighbor = (row - offset) + irow;
297  if (rowNeighbor < 0 || rowNeighbor >= hDet->GetNbinsY())
298  continue; // avoids under and overflow
299  for (int icol = 0; icol < nbuffer; icol++) {
300  if (icol == 1 && irow == 1)
301  continue; // exclude self
302  int colNeighbor = (column - offset) + icol;
303  if (colNeighbor < 0 || colNeighbor >= hDet->GetNbinsX())
304  continue; // avoids under and overflow
305  float tempGain = hDet->GetBinContent(hDet->GetXaxis()->FindBin(colNeighbor), hDet->GetYaxis()->FindBin(rowNeighbor));
306  if (tempGain <= 0)
307  continue; // exclude empty/negative bins
308  gainNeighbor.push_back(tempGain);
309  }
310  }
311  int numberOfNeighbors = (int)gainNeighbor.size();
312  if (numberOfNeighbors < 1)
313  return 0; // if empty, return 0
314 
315  average = std::accumulate(gainNeighbor.begin(), gainNeighbor.end(), decltype(gainNeighbor)::value_type(0.0)) / numberOfNeighbors;
316 
317  return average;
318 }

347 bool PadCalibCCDBBuilder::IsHotAreaIsolated(TH2F* hDet, int column, int row, int matrixSize)
348 {
349  bool isIsolated = kTRUE;
350  float averageGain = ComputeDetectorAverage(hDet);
351  int nSurroundingNotHot = 0;
352  int nMaxHot = 8 + (matrixSize - 1) * 4;
353  float averageAround = 0.;
354  int nUsedBins = 0;
355  for (int i = 0; i < matrixSize + 2; i++) {
356  int icol = i - 1;
357  for (int j = 0; j < matrixSize + 2; j++) {
358  int jrow = j - 1;
359  if ((-1 < icol) && (icol < matrixSize) && (-1 < jrow) && (jrow < matrixSize))
360  continue;
361  if (abs(icol) <= (int)(matrixSize - 1) / 2)
362  continue;
363  float temp = hDet->GetBinContent(column + icol, row + jrow);
364  if (temp <= 2 * averageGain && temp != 0) {
365  nUsedBins++;
366  averageAround += temp;
367  } else if (temp == 0) {
368  nSurroundingNotHot++;
369  }
370  }
371  }
372 
373  averageAround /= nUsedBins;
374  if (averageAround < 2 * averageGain && nSurroundingNotHot < nMaxHot)
375  isIsolated = kFALSE;
376 
377  return isIsolated;
378 }
